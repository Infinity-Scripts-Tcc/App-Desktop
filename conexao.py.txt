import mysql.connector

# Configuração das credenciais do banco de dados
host = "127.0.0.1"
user = "root"
password = ""
database = "glam"  # Coloque o nome do seu banco de dados

try:
    # Estabelece a conexão
    connection = mysql.connector.connect(
        host=host,
        user=user,
        password=password,
        database=database
    )

    # Verifica se a conexão foi bem-sucedida
    if connection.is_connected():
        print("Conexão ao banco de dados MySQL estabelecida com sucesso!")
    else:
        print("Não foi possível conectar ao banco de dados.")

    # Cria o cursor
    cursor = connection.cursor()

    # Consulta SQL
    sql = "SELECT * FROM NOME_DA_TABELA ORDER BY CAMPO_DA_CHAVE_PRIMARIA DESC"

    # Executa a consulta
    cursor.execute(sql)

    # Obtém os resultados
    results = cursor.fetchall()

    # Itera pelos resultados
    for row in results:
        nome_do_campo = row["nome_do_campo"]
        print(nome_do_campo)

    # Exemplo de uso específico
    pegando = results[0]["nome_do_campo"]
    print(f"Valor de 'nome_do_campo': {pegando}")

except mysql.connector.Error as err:
    print(f"Erro ao conectar ao banco de dados ou executar a consulta: {err}")

finally:
    # Fecha o cursor e a conexão quando terminar
    cursor.close()
    connection.close()
